# 1018锤子剪刀布

```c++
#include<bits/stdc++.h>
using namespace std;

int papi(char a,char b){
	if(a==b) return 0;
	else if(a=='B'&&b=='C') return 1;
	else if(a=='C'&&b=='J') return 1;
	else if(a=='J'&&b=='B') return 1; 
	else return -1;
} 

int main(){
	int total;
	cin>>total;
	int pingju=0;
	int jw=0;
	int jl=0;
	int aB=0,aC=0,aJ=0,bB=0,bC=0,bJ=0;
	for(int i=0;i<total;i++){
		char a,b;
		cin>>a>>b;
		if(papi(a,b)==0) pingju++;
		else if(papi(a,b)==1){
			jw++;
			if(a=='C') aC++;
			else if(a=='B') aB++;
			else if(a=='J') aJ++;
		}
		else if(papi(a,b)==-1){
			jl++;
			if(b=='C') bC++;
			else if(b=='B') bB++;
			else if(b=='J') bJ++;
		}
	}
	//cout<<"aB="<<aB<<"aJ="<<aJ<<endl;
	cout<<jw<<" "<<pingju<<" "<<jl<<endl<<jl<<" "<<pingju<<" "<<jw<<endl;
	char max;
	if(aB>=aC&&aB>=aJ) max='B';
	else if(aC>=aB&&aC>=aJ) max='C';
	else max='J';
	cout<<max<<" ";
	if(bB>=bC&&bB>=bJ) max='B';
	else if(bC>=bB&&bC>=bJ) max='C';
	else max='J';
	cout<<max;
	return 0;
}
```
